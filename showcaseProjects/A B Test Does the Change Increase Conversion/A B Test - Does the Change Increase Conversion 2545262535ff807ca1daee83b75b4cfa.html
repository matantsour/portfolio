    • If the p-value is greater than or equal to 0.05:

      We fail to reject H0 . This means the data do not provide sufficient evidence that the 7-day trial improves paid conversion rates; any observed difference could be due to chance.

A) Calculating measures script for A/B test:

WITH users_tbl AS (
  SELECT * 
  FROM `data_q1.t1_userbase`
  WHERE exp_id = 'ab_test_trial_purchase'
),

duplicate_users AS (
  SELECT userid
  FROM users_tbl
  GROUP BY userid
  HAVING COUNT(*) > 1
),

users_no_dups AS (
  SELECT u1.*
  FROM users_tbl u1
  LEFT JOIN duplicate_users u2 ON u1.userid = u2.userid
  WHERE u2.userid IS NULL
),

transactions_tbl AS (
  SELECT 
    ruserid,
    event,
    plan,
    timestamp,
    price
  FROM `data_q1.t2_transactions`
),

duplicate_transactions AS (
  SELECT 
    ruserid,
    event, 
    plan
  FROM transactions_tbl
  GROUP BY ruserid, event, plan
  HAVING COUNT(*) > 1
),

transactions_no_dups AS (
  SELECT t1.*
  FROM transactions_tbl t1
  LEFT JOIN duplicate_transactions t2 
    ON t1.ruserid = t2.ruserid 
    AND t1.event = t2.event 
    AND t1.plan = t2.plan
  WHERE t2.ruserid IS NULL
),

transaction_users AS (
  SELECT 
    *
  FROM transactions_no_dups AS T
  JOIN users_no_dups AS U
    ON T.ruserid = U.userid
),

users_per_group AS (
  SELECT
    test_group_flag,
    COUNT(DISTINCT userid) AS total_users
  FROM transaction_users
  GROUP BY test_group_flag
),

purchase_users AS (
  SELECT DISTINCT
    userid,
    test_group_flag
  FROM transaction_users
  WHERE event = 'purchase'
    AND price > 0
),

refund_users AS (
  SELECT DISTINCT
    userid,
    test_group_flag
  FROM transaction_users
  WHERE event = 'refund'
),

converted_users AS (
  SELECT
    p.userid,
    p.test_group_flag
  FROM purchase_users p
  LEFT JOIN refund_users r ON p.userid = r.userid
  WHERE r.userid IS NULL
),

refund_count_per_group AS (
  SELECT
    test_group_flag,
    COUNT(DISTINCT userid) AS refund_users
  FROM refund_users
  GROUP BY test_group_flag
)

SELECT
  u.test_group_flag,
  u.total_users,
  COUNT(DISTINCT c.userid) AS converted_users,
  COALESCE(r.refund_users, 0) AS refund_users,
  ROUND(COUNT(DISTINCT c.userid) * 1.0 / u.total_users,4) AS paid_conversion_rate
FROM users_per_group u
LEFT JOIN converted_users c ON u.test_group_flag = c.test_group_flag
LEFT JOIN refund_count_per_group r ON u.test_group_flag = r.test_group_flag
GROUP BY u.test_group_flag, u.total_users, r.refund_users
ORDER BY u.test_group_flag;

metrics AS (
  SELECT
    COALESCE(CAST(test_group_flag AS STRING), 'ALL_GROUPS') AS group_flag,
    COALESCE(plan, 'ALL_PLANS') AS plan_type,
    COUNT(DISTINCT userid) AS total_users,

    COUNT(DISTINCT CASE WHEN event = 'refund' THEN userid END) AS total_refund_users,
    COUNT(DISTINCT CASE WHEN event = 'purchase' AND price > 0 THEN userid END)
      - COUNT(DISTINCT CASE WHEN event = 'refund' THEN userid END) AS actual_paying_users,
    ROUND(SAFE_DIVIDE(COUNT(DISTINCT CASE WHEN event = 'refund' THEN userid END),
          COUNT(DISTINCT CASE WHEN event = 'purchase' AND price > 0 THEN userid END)), 4) AS refund_rate_pct,
    ROUND(SAFE_DIVIDE(
          COUNT(DISTINCT CASE WHEN event = 'purchase' AND price > 0 THEN userid END)
          - COUNT(DISTINCT CASE WHEN event = 'refund' THEN userid END),
          COUNT(DISTINCT userid)), 4) AS paid_conversion_rate_pct,
    ROUND(SUM(CASE WHEN event = 'purchase' THEN price ELSE 0 END), 2) AS total_income,

    ROUND(SUM(CASE WHEN event = 'refund' THEN price ELSE 0 END), 2) AS total_refund_amount,

      - SUM(CASE WHEN event = 'refund' THEN price ELSE 0 END), 2) AS total_revenue,
    ROUND(SAFE_DIVIDE(
          SUM(CASE WHEN event = 'purchase' THEN price ELSE 0 END) 
          - SUM(CASE WHEN event = 'refund' THEN price ELSE 0 END),
          COUNT(DISTINCT CASE WHEN event = 'purchase' AND price > 0 AND userid NOT IN

    SUM(actual_paying_users) AS actual_paying_users,
    ROUND(SAFE_DIVIDE(SUM(total_refund_users), SUM(total_paying_users)), 4) AS refund_rate_pct,
    ROUND(SAFE_DIVIDE(SUM(actual_paying_users), SUM(total_users)), 4) AS paid_conversion_rate_pct,
    ROUND(SUM(total_income), 2) AS total_income,
    ROUND(SAFE_DIVIDE(SUM(total_income), SUM(total_paying_users)), 2) AS avg_income,
    ROUND(SUM(total_refund_amount), 2) AS total_refund_amount,
    ROUND(SAFE_DIVIDE(SUM(total_refund_amount), SUM(total_refund_users)), 2) AS avg_refund_amount,
    ROUND(SUM(total_revenue), 2) AS total_revenue,
    ROUND(SAFE_DIVIDE(SUM(total_revenue), SUM(actual_paying_users)), 2) AS avg_revenue
  FROM metrics
  GROUP BY plan_type
),

-- 4) Per group + plan
per_group_plan AS (
  SELECT 'PER_GROUP_PLAN' AS analysis_level, 
    CASE when group_flag = '0' then 'No Trial Period' else 'Trial Period' end as group_flag,
    plan_type,
    SUM(total_users) AS total_users,
    SUM(total_paying_users) AS total_paying_users,
    SUM(total_refund_users) AS total_refund_users,
    SUM(actual_paying_users) AS actual_paying_users,
    ROUND(SAFE_DIVIDE(SUM(total_refund_users), SUM(total_paying_users)), 4) AS refund_rate_pct,
    ROUND(SAFE_DIVIDE(SUM(actual_paying_users), SUM(total_users)), 4) AS paid_conversion_rate_pct,
    ROUND(SUM(total_income), 2) AS total_income,
    ROUND(SAFE_DIVIDE(SUM(total_income), SUM(total_paying_users)), 2) AS avg_income,
    ROUND(SUM(total_refund_amount), 2) AS total_refund_amount,
    ROUND(SAFE_DIVIDE(SUM(total_refund_amount), SUM(total_refund_users)), 2) AS avg_refund_amount,
    ROUND(SUM(total_revenue), 2) AS total_revenue,
    ROUND(SAFE_DIVIDE(SUM(total_revenue), SUM(actual_paying_users)), 2) AS avg_revenue
  FROM metrics
  GROUP BY group_flag, plan_type
)

-- Combine all analyses
SELECT * FROM total_pop
UNION ALL
SELECT * FROM per_group
UNION ALL 
SELECT * FROM per_plan
UNION ALL
SELECT * FROM per_group_plan
ORDER BY 
CASE analysis_level
WHEN 'PER_GROUP' THEN 1
WHEN 'PER_PLAN' THEN 2
WHEN 'PER_GROUP_PLAN' THEN 3
WHEN 'TOTAL_POPULATION' THEN 4
END,
group_flag, plan_type;
